🏗️ 10. Class và Object

Class = Bản vẽ thiết kế xe hơi
Object = Chiếc xe thật được đúc ra
🐍 Python
class ConMeo:
    def __init__(self, ten, mau):
        self.ten = ten
        self.mau = mau
    def keu(self):
        print(f"{self.ten}: Meo!")
 
meo1 = ConMeo("Mun","Đen")
meo1.keu()
🌐 JavaScript
class ConMeo {
    constructor(ten, mau) {
        this.ten = ten;
        this.mau = mau;
    }
    keu() {
        console.log(this.ten+": Meo!");
    }
}
let m = new ConMeo("Mun","Đen");
m.keu();

// Scope: biến chỉ sống trong { } của nó

# Try/Catch - Dự phòng sự cố
try:
    a = 10 / 0
except ZeroDivisionError:
    print("Không chia được 0!")
finally:
    print("Luôn chạy dù có lỗi")
// Async/Await - Lời hứa bất đồng bộ
async function getData() {
    try {
        const res = await fetch('https://api.example.com/data');
        const data = await res.json();
        console.log(data);
    } catch (e) { console.log("Lỗi mạng:", e); }
}

🐍 Echo Bot v20+

# bot.py
from telegram import Update
from telegram.ext import (
    Application, CommandHandler,
    MessageHandler, filters
)
TOKEN = "YOUR_BOT_TOKEN"
 
async def start(update, ctx):
    await update.message.reply_text(
        "👋 Chào! Gõ gì đó đi!")
 
async def echo(update, ctx):
    await update.message.reply_text(
        f"Bạn nói: {update.message.text}")
 
def main():
    app=Application.builder().token(TOKEN).build()
    app.add_handler(CommandHandler("start",start))
    app.add_handler(MessageHandler(
        filters.TEXT&~filters.COMMAND, echo))
    app.run_polling()
 
if __name__=="__main__": main()
